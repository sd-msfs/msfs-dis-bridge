25 std::cerr << "[Multicast] WSAStartup failed\n";

27 return false;

32 if (sock_ == INVALID_SOCKET) {

34 running_ = false;
35 WSACleanup();
36 return false;
37 }
38
39 // Optional: bigger send buffer
40 int snd = 1<<20; // 1 MB
41 ::setsockopt(sock_, SOL_SOCKET, SO_SNDBUF, reinterpret_cast<const char*>(&snd), sizeof(snd));
42
43 // TTL + loopback
44 unsigned char ttl_uc = static_cast<unsigned char>(ttl);
45 ::setsockopt(sock_, IPPROTO_IP, IP_MULTICAST_TTL, reinterpret_cast<const char*>(&ttl_uc), sizeof(ttl_uc));

48

53 ::setsockopt(sock_, IPPROTO_IP, IP_MULTICAST_IF, reinterpret_cast<const char*>(&iface), sizeof(iface));
54 } else {
55 std::cerr << "[Multicast] Invalid interface IP: " << ifaddr << "\n";
56 }
57 }
58
59 // (Nice-to-have) suppress UDP “connection reset” noise
60 DWORD bytes=0, newBehavior = FALSE;
61 ::WSAIoctl(sock_, SIO_UDP_CONNRESET, &newBehavior, sizeof(newBehavior), nullptr, 0, &bytes, nullptr, nullptr);
62

64 memset(&dst_, 0, sizeof(dst_));

70 running_ = false;
71 WSACleanup();
72 return false;
73 }

76 port_ = port;

78 std::cout << "[Multicast] Sending to " << group_ << ":" << port_

82 sender_ = std::thread(&UDPMulticaster::senderLoop, this);
