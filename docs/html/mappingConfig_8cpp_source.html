10bool MappingConfig::loadProfileFromCSV(const std::string& path) {
11 mappings_.clear();

17
18 std::string line;

23
24 if (!std::getline(ss, eventName, ',')) continue;
25 if (!std::getline(ss, fieldName, ',')) continue;
26 if (!std::getline(ss, pduType, ',')) continue;
27 if (!std::getline(ss, pduField, ',')) continue;
28 if (!std::getline(ss, enabledStr, ',')) enabledStr = "true";
29 if (!std::getline(ss, rateStr, ',')) rateStr = "event";
30
31 MappingEntry entry;
32 entry.eventName = eventName;
33 entry.fieldName = fieldName;
34 entry.pduType = pduType;
35 entry.pduField = pduField;
36 entry.enabled = (enabledStr == "true" || enabledStr == "1");
37
38 if (rateStr == "event")
39 entry.rateHz = 0.0;
40 else
41 entry.rateHz = std::stod(rateStr);
42
43 entry.lastSent = std::chrono::steady_clock::now();
44
45 mappings_.push_back(entry);
46 }
47 return true;
48}
