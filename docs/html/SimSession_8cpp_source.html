34bool SimSession::start() {
35 running_ = true;
36
37 // Create a Windows event that SimConnect will signal
38 evt_ = CreateEvent(nullptr, FALSE, FALSE, nullptr);
39 if (!evt_) {
40 std::cerr << "[" << name_ << "] Failed to create event handle\n";
41 return false;
42 }
43
44 // Open SimConnect using the given ConfigIndex
45 HRESULT hr = SimConnect_Open(&hSim_, name_.c_str(), nullptr, 0, evt_, cfgIndex_);
46 if (hr != S_OK) {
47 std::cerr << "[" << name_ << "] SimConnect_Open failed (cfgIndex=" << cfgIndex_ << ")\n";
48 return false;
49 }
50
51 // Subscribe to pause/sim events
52 SimConnect_SubscribeToSystemEvent(hSim_, EVT_PAUSE_EX1, "Pause_EX1");
53 SimConnect_SubscribeToSystemEvent(hSim_, EVT_SIM, "Sim");
54
55 // Register the data definition and request updates
56 defineData_();
57 requestStream_();
58
59 // Spawn the worker thread
60 th_ = std::thread([this] { runLoop_(); });
61 return true;
62}

88 while (running_) {
89 DWORD wait = WaitForSingleObject(evt_, 100);

91 SimConnect_CallDispatch(hSim_, &dispatchThunk, this);

120// Handle incoming messages
121void SimSession::onDispatch_(SIMCONNECT_RECV* pData, DWORD) {
122 switch (pData->dwID) {
123 case SIMCONNECT_RECV_ID_EVENT: {
124 auto* e = reinterpret_cast<SIMCONNECT_RECV_EVENT*>(pData);
125 if (e->uEventID == EVT_PAUSE_EX1) {
126 uint32_t flags = static_cast<uint32_t>(e->dwData);
127 pauseFlags_.store(flags, std::memory_order_relaxed);
128 } else if (e->uEventID == EVT_SIM) {
129 // 1 = running, 0 = not running
130 simRunning_.store(static_cast<uint32_t>(e->dwData), std::memory_order_relaxed);
131 }
132 break;
133 }
134
135 case SIMCONNECT_RECV_ID_SIMOBJECT_DATA: {
136 auto* d = reinterpret_cast<SIMCONNECT_RECV_SIMOBJECT_DATA*>(pData);
137 if (d->dwRequestID == REQ_ID && !isPaused()) {
138 const SimSample* s = reinterpret_cast<const SimSample*>(&d->dwData);
139
140 // print out to console for debugging
141 if (doPrint) {
142 std::cout << "[" << name_ << "] Lat=" << s->lat
143 << " Lon=" << s->lon
144 << " Alt(m)=" << s->alt
145 << " As(kts)=" << s->airspeed
146 << "\n";
147 }
148
149 // --- convert to FlightData expected by your encoder ---
150 FlightData fd{};
151 fd.latitude = s->lat;
152 fd.longitude = s->lon;
153 // Your encoder expected feet in your earlier example; convert from meters:
154 fd.altitude = s->alt * 3.280839895; // m -> ft
155 fd.pitch = s->pitch;
156 fd.bank = s->bank;
157 fd.heading = s->heading;
158 fd.airspeed = s->airspeed;
159
160 // --- encode via your existing mapping/encoder ---
161 std::vector<std::uint8_t> packet = g_encoder.encodeEvent(fd);
162
163 // --- enqueue to multicast sender (non-blocking) ---
164 UDPMulticaster::getInstance().enqueue(std::move(packet));
165 }
166 break;
167 }
168
169 default:
170 break;
171 }
