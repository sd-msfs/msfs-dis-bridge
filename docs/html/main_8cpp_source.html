27static const std::string base64_chars =
28"ABCDEFGHIJKLMNOPQRSTUVWXYZ"
29"abcdefghijklmnopqrstuvwxyz"
30"0123456789+/";

32static std::string base64_encode(const std::vector<uint8_t>& bytes) {
33 std::string ret;
34 int val = 0, valb = -6;
35 for (uint8_t c : bytes) {
36 val = (val << 8) + c;
37 valb += 8;
38 while (valb >= 0) {
39 ret.push_back(base64_chars[(val >> valb) & 0x3F]);
40 valb -= 6;
41 }
42 }
43 if (valb > -6) ret.push_back(base64_chars[((val << 8) >> (valb + 8)) & 0x3F]);
44 while (ret.size() % 4) ret.push_back('=');
45 return ret;
46}

64void CALLBACK MyDispatchProc(SIMCONNECT_RECV* pData, DWORD cbData, void* pContext) {
65 switch (pData->dwID) {
66 case SIMCONNECT_RECV_ID_SIMOBJECT_DATA: {
67 auto* pObjData = (SIMCONNECT_RECV_SIMOBJECT_DATA*)pData;
68 if (pObjData->dwRequestID == 1) {
69 FlightData* fd = reinterpret_cast<FlightData*>(&pObjData->dwData);
70
71 bool sent = false;
72 auto* entry = mappingConfig.getMapping("FlightDataUpdate");
73
74 if (entry && entry->enabled) {
75 // Rate-limited send
76 auto now = std::chrono::steady_clock::now();
77 double intervalMs = 1000.0 / std::max(1.0, entry->rateHz);
78 auto it = lastSent.find(entry->eventName);
79
80 if (it == lastSent.end() ||
81 std::chrono::duration_cast<std::chrono::milliseconds>(now - it->second).count() >= intervalMs)
82 {
83 lastSent[entry->eventName] = now;
84
85 auto packet = encoder.encodeEvent(*fd);
86 sendto(udpSocket,
87 reinterpret_cast<const char*>(packet.data()),
88 static_cast<int>(packet.size()), 0,
89 reinterpret_cast<sockaddr*>(&dest),
90 sizeof(dest));
91
92 std::cout << "[DIS Bridge] Sent " << entry->pduType
93 << " for event " << entry->eventName
94 << " lat=" << fd->latitude
95 << " lon=" << fd->longitude << std::endl;
96
97 sent = true;
98 }
99 }
100
101 if (!sent) {
102 // Fallback behavior (no mapping loaded or mapping disabled)
103 auto packet = encoder.encodeEvent(*fd);
104 sendto(udpSocket,
105 reinterpret_cast<const char*>(packet.data()),
106 static_cast<int>(packet.size()), 0,
107 reinterpret_cast<sockaddr*>(&dest),
108 sizeof(dest));

110 std::cout << "[DIS Bridge] (Fallback) lat=" << fd->latitude
111 << " lon=" << fd->longitude << std::endl;
112 }
113 }
114 break;
115 }
116 default:
117 break;
118 }
